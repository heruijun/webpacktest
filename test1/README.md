# 项目里安装webpack
    npm install webpack -D 
    或
    yarn add webpack -D

# 安装webpack-cli
    yarn add webpack webpack-cli -D

# 安装webpack静态服务器依赖
    yarn add webpack webpack-dev-server -D

# 安装html-webpack-plugin
    yarn add html-webpack-plugin -D

# 如果需要单独把 CSS 文件分离出来，我们需要使用 extract-text-webpack-plugin 插件
